A=\sum_{n=1}^\infty A_n

\end{tcolorbox}
\begin{tcolorbox}[arc=0pt,colback=white,title={Problem}]
Given $n\in\mathbb{N}\cup\{0\}$, find the $(n-1)$-th to $(n+8)$-th terms of $A$?
